listado de productos con su categoria:
SELECT c.nombre, p.nombre FROM `productos` p 
inner join categorias c
on p.idcategoria = c.id
order by 1,2;

Categorias que no tienen productos
SELECT c.nombre, p.nombre FROM categorias c
left join productos p
on p.idcategoria = c.id
where p.id is null;

Vale tambien:
SELECT c.nombre, p.nombre FROM productos p
right join categorias c
on p.idcategoria = c.id
where p.id is null;

SELECT nombre from categorias
where id not in (select DISTINCT idcategoria from productos);
